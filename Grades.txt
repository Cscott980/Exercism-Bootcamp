function build_url with string do
    set list to my#split(string, " ")
    set version to "v4"
        if my#string_to_number(list[4]) < 2025 do 
            change version to "v3"
        end       
    return concatenate("https://api.school.com/", version, "/grades/", list[4], "/class-",list[2])
end
function grades_to_pattern with url do
    set new_dictionary to {}
    set new_url to build_url(url)
    set dictionary to fetch(new_url,{})

        if my#length(my#split(dictionary["data"]["teacher"]," ")) > 2 do
            change new_dictionary["teacher"] to concatenate(my#capitalize(my#split(dictionary["data"]["teacher"]," ")[2])," ",my#capitalize(my#split(dictionary["data"]["teacher"]," ")[3]))
        else do
            change new_dictionary["teacher"] to my#capitalize(my#split(dictionary["data"]["teacher"]," ")[2])
        end

    change new_dictionary["grades"] to my#sort_string(dictionary["data"]["grades"])
    return new_dictionary
end
